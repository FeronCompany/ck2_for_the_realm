

ftr_expand_influence_prestige_cost_value = 500

ftr_expand_influence_value = 20
ftr_shift_focus_cost_value = 80
ftr_shift_focus_modify_value = -80

ftr_payment_cost_level_value = {
	value = 0
	if = {
		limit = {
			scope:half_price = yes
		}
		add = 0.5
	}
	else_if = {
		limit = {
			scope:normal_price = yes
		}
		add = 1
	}
	else_if = {
		limit = {
			scope:double_price = yes
		}
		add = 2
	}
	else_if = {
		limit = {
			scope:three_times_price = yes
		}
		add = 3
	}
	else_if = {
		limit = {
			scope:four_times_price = yes
		}
		add = 4
	}
	else_if = {
		limit = {
			scope:five_times_price = yes
		}
		add = 5
	}
}

ftr_buy_favor_cost_value = {
	value = 0
	add = {
		value = gift_value
		multiply = ftr_payment_cost_level_value
	}
}

			multiply = 500

	else_if = {
		limit = {
			current_date <= 1100.1.1
		}
		add = {
			value = scope:secondary_recipient.sub_realm_size
			multiply = 600
		}
	}
	else_if = {
		limit = {
			current_date <= 1200.1.1
		}
		add = {
			value = scope:secondary_recipient.sub_realm_size
			multiply = 700
		}
	}
	else_if = {
		limit = {
			current_date <= 1300.1.1
		}
		add = {
			value = scope:secondary_recipient.sub_realm_size
			multiply = 800
		}
	}
	else = {
