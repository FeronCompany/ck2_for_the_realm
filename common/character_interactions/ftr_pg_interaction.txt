
######### Bribe ##########
ftr_bribe_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	cooldown_against_recipient = { years = 5 }
	desc = ftr_bribe_interaction_desc
	greeting = positive
	notification_text = ftr_bribe_interaction_notification
	ai_targets = {
		ai_recipients = liege
		ai_recipients = peer_vassals
		ai_recipients = vassals
		max = 3
	}
	ai_frequency = 24
	#ai_maybe = yes
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			is_landed = yes
		}
		OR = {
			scope:actor.liege = scope:recipient
			scope:recipient = {
				liege = scope:actor
				is_landed = yes
			}
			AND = {
				scope:actor = {
					is_independent_ruler = no
				}
				scope:actor.liege = scope:recipient.liege
				scope:recipient = {
					OR = {
						is_landed = yes
						is_councillor = yes
					}
				}
			}
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT =  { is_at_war_with = scope:recipient }
			is_imprisoned = no
			is_capable_adult = yes
			custom_description = {
				text = ftr_enough_gold_tt
				value = gift_value
				gold >= gift_value
			}
		}
		scope:recipient = {
			is_imprisoned = no
			is_capable_adult = yes
		}
	}
	on_accept = {
		scope:actor = {
			pay_short_term_gold = {
				gold = gift_value
				target = scope:recipient
			}
			ftr_add_political_influence_effect = {
				VALUE = ftr_bribe_influence_gain_value
			}
			trigger_event = ftr_pg_interaction_event.0001
		}
	}
	on_decline = {}
	#auto_accept = yes
	ai_accept = {
		base = -40
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
			desc = AI_OPINION_REASON
		}
	}
	ai_will_do = {
		base = -40
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 0.25
			desc = AI_OPINION_REASON
		}
		modifier = {
			add = -20
			scope:actor = {
				has_trait = greedy
			}
		}
		modifier = {
			add = 40
			scope:actor = {
				gold >= gift_value
				ftr_political_influence_value < 40
			}
		}
		modifier = {
			add = 10
			scope:actor.liege = scope:recipient
		}
		modifier = {
			add = 10
			scope:actor.liege = scope:recipient.liege
			scope:recipient = { is_councillor = yes }
		}
	}
}

	common_interaction = yes

	ai_targets = {
		ai_recipients = peer_vassals
		max = 2
	}
	ai_frequency = 36
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			liege = scope:recipient.liege
			is_independent_ruler = no
		}
		scope:recipient = {
			is_independent_ruler = no
		}
		scope:actor.liege = {
			has_imprisonment_reason = scope:recipient
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			is_imprisoned = no
			is_capable_adult = yes

				text = "ftr_have_enough_pi_tt"
				value = ftr_charge_cost_value

			}
		}
		scope:recipient = {
			is_imprisoned = no
			is_capable_adult = yes
			is_at_war = no
		}
	}
	on_accept = {
		scope:actor = {
			ftr_add_political_influence_effect = {
				VALUE = ftr_charge_modify_value
			}
		}
		scope:recipient = {
			add_opinion = {
				modifier = angry_opinion
				target = scope:actor
				opinion = -60
			}
		}
		scope:actor.liege = {
			trigger_event = ftr_pg_interaction_event.0100
		}
	}
	on_decline = {}
	auto_accept = yes

		base = -20
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient

			desc = AI_OPINION_REASON
		}
	}

######### Sow Discord ##########
ftr_sow_discord_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	cooldown = { years = 2 }
	cooldown_against_recipient = { years = 5 }
	desc = ftr_sow_discord_interaction_desc
	ai_targets = {
		ai_recipients = peer_vassals
		ai_recipients = neighboring_rulers
		max = 3
	}
	ai_frequency = 64
	is_shown = {
		scope:actor = {
			OR = {
				intrigue >= decent_skill_rating
				has_trait = deceitful
			}
		}
		OR = {
			scope:actor = { is_vassal_of = scope:recipient }
			scope:recipient = { is_vassal_of = scope:actor }
			scope:actor = {
				is_landed = yes
				is_independent_ruler = yes
				character_is_realm_neighbor = scope:recipient
			}
			scope:actor = {
				liege = scope:recipient.liege
				NOT = { this = scope:recipient.liege }
				is_landed = yes
			}
		}
		scope:recipient = {
			primary_title.tier >= tier_county
			is_landed = yes
			NOT = { this = scope:actor }
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			is_imprisoned = no
			is_capable_adult = yes
			custom_description = {
				text = "ftr_have_enough_pi_tt"
				value = ftr_sow_discord_cost_value
				ftr_political_influence_value >= ftr_sow_discord_cost_value
			}
		}
		scope:recipient = {
			is_imprisoned = no
			is_capable_adult = yes
		}
	}
	populate_actor_list = {
		scope:recipient = {
			every_vassal = {
				limit = {
					is_capable_adult = yes
					NOT = { scope:actor = this }
				}
				add_to_list = characters
			}
			every_courtier = {
				limit = {
					is_capable_adult = yes
					NOT = { scope:actor = this }
				}
				add_to_list = characters
			}
			if = {
				limit = { is_independent_ruler = no }
				liege = {
					if = {
						limit = { NOT = { scope:actor = this } }
						add_to_list = characters
					}
					every_vassal = {
						limit = {
							is_capable_adult = yes
							NOT = { scope:actor = this }
						}
						add_to_list = characters
					}
				}
			}
		}
	}
	can_be_picked = {
		NOT = { this = scope:recipient }
		NOT = { has_relation_rival = scope:recipient }
		NOT = { has_relation_nemesis = scope:recipient }
	}

	on_accept = {
		scope:actor = {
			ftr_add_political_influence_effect = {
				VALUE = ftr_sow_discord_modify_value
			}
			stress_impact = {
				just = minor_stress_impact_gain
				honest = minor_stress_impact_gain
				deceitful = minor_stress_impact_loss
			}
		}
		custom_tooltip = {
			text = ftr_sow_discord_desc_text
		}
		random_list = {
			10 = { # Sow discord success
				modifier = {
					add = scope:actor.intrigue
				}
				scope:recipient = {
					trigger_event = {
						id = ftr_pg_interaction_event.0200
						days = { 15 30 }
					}
				}
				desc = ftr_sow_discord.success
			}
			10 = { # Failed but not discovered
				modifier = {
					add = scope:recipient.intrigue
					scope:recipient.intrigue < high_skill_rating
				}
				scope:recipient = {
					trigger_event = {
						id = ftr_pg_interaction_event.0201
						days = { 15 30 }
					}
				}
				desc = ftr_sow_discord.failed
			}
			0 = { # Failed and discovered
				modifier = {
					add = scope:recipient.intrigue
					scope:recipient.intrigue >= high_skill_rating
				}
				scope:recipient = {
					trigger_event = {
						id = ftr_pg_interaction_event.0203
						days = { 15 30 }
					}
				}
				desc = ftr_sow_discord.discovered
			}
		}
	}
	on_decline = {}
	auto_accept = yes
	ai_will_do = {
		base = 0
		opinion_modifier = {
			opinion_target = scope:recipient
			who = scope:actor
			multiplier = -1.5
		}
		opinion_modifier = {
			opinion_target = scope:secondary_actor
			who = scope:actor
			multiplier = -0.5
		}
		modifier = {
			add = -50
			OR = {
				has_trait = just
				has_trait = honest
				has_trait = content
			}
		}
		modifier = {
			add = 20
			has_trait = deceitful
		}
		modifier = {
			add = 20
			scope:actor = {
				opinion = {
					target = scope:recipient
					value < 30
				}
				is_vassal_of = scope:secondary_actor
			}
			scope:recipient = {
				is_vassal_of = scope:secondary_actor
			}
		}
	}
}

######### Support Heir ##########
ftr_support_heir_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	cooldown = { years = 2 }
	cooldown_against_recipient = { years = 10 }
	desc = ftr_support_heir_interaction_desc
	is_shown = {
		scope:actor = {
			is_independent_ruler = no
			liege = scope:recipient.liege
		}
		scope:recipient = {
			is_independent_ruler = no
			OR = {
				is_heir_of = liege
				is_child_of = liege
			}
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			is_imprisoned = no
			is_capable_adult = yes
			custom_description = {
				text = "ftr_have_enough_pi_tt"
				value = ftr_support_heir_cost_value
				ftr_political_influence_value >= ftr_support_heir_cost_value
			}
		}
		scope:recipient = {
			is_imprisoned = no
			NOT = { has_trait = disinherited }
			NOT = { has_trait = disputed_heritage }
			NOT = { has_trait = denounced }
		}
	}

	options_heading = ftr_support_heir_options_tt
	send_option = { # Support in secret
		flag = ftr_support_in_secret
		localization = ftr_support_in_secret_tt
		starts_enabled = { always = yes	}
	}
	send_option = { # Support in public
		flag = ftr_support_in_public
		localization = ftr_support_in_public_tt
	}
	send_options_exclusive = yes

	on_accept = {
		scope:actor = {
			ftr_add_political_influence_effect = {
				VALUE = ftr_support_heir_modify_value
			}
			liege = {
				save_scope_as = liege_c
				every_heir = {
					limit = { NOT = { this = scope:recipient } }
					scope:actor = {
						add_opinion = {
							target = prev
							modifier = ftr_political_opponent_opinion
						}
					}
				}
			}
		}
		scope:recipient = {
			add_prestige = 200
			add_opinion = {
				target = scope:actor
				modifier = ftr_political_alliance_opinion
			}
		}
		if = {
			limit = { scope:ftr_support_in_public = yes }
			scope:actor.liege = {
				every_heir = {
					limit = { NOT = { this = scope:recipient } }
					add_opinion = {
						target = scope:actor
						modifier = ftr_political_opponent_opinion
					}
				}
				trigger_event = {
					id = ftr_pg_interaction_event.0303
					days = 5
				}
			}
			scope:recipient = {
				ftr_add_political_influence_effect = {
					VALUE = ftr_support_heir_cost_value
				}
				trigger_event = ftr_pg_interaction_event.0300
			}
		}
		else = {
			scope:recipient = {
				trigger_event = ftr_pg_interaction_event.0301
			}
		}
	}
	on_decline = {}
	auto_accept = yes
	ai_will_do = {
		base = -30
		opinion_modifier = {
			opinion_target = scope:recipient
			who = scope:actor
			multiplier = 0.5
		}
		modifier = {
			add = -50
			scope:actor = { is_heir_of = liege }
		}
		modifier = {
			add = 20
			scope:actor = {
				is_allied_to = scope:recipient
			}
		}
		modifier = {
			add = 20
			scope:actor = {
				has_trait = ambitious
			}
		}
		modifier = {
			add = -30
			scope:actor = {
				has_trait = content
				has_trait = humble
			}
		}
		modifier = {
			add = 20
			scope:actor.liege = {
				has_active_diarchy = yes
			}
		}
		modifier = {
			add = -30
			scope:actor = {
				has_character_flag = ftr_liege_warn_not_support_heir
			}
		}
	}
}
