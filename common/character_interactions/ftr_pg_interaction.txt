
######### Bribe ##########
ftr_bribe_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	cooldown_against_recipient = { years = 5 }
	desc = ftr_bribe_interaction_desc
	ai_targets = {
		ai_recipients = liege
		ai_recipients = peer_vassals
		max = 3
	}
	ai_frequency = 12
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			is_landed = yes
		}
		OR = {
			scope:actor.liege = scope:recipient
			scope:recipient = {
				liege = scope:actor
				is_landed = yes
			}
			AND = {
				scope:actor = {
					is_independent_ruler = no
				}
				scope:actor.liege = scope:recipient.liege
				scope:recipient = {
					OR = {
						is_landed = yes
						is_councillor = yes
					}
				}
			}
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT =  { is_at_war_with = scope:recipient }
			is_imprisoned = no
			is_capable_adult = yes
			gold > gift_value
		}
		scope:recipient = {
			is_imprisoned = no
			is_capable_adult = yes
		}
	}
	on_accept = {
		scope:actor = {
			pay_short_term_gold = {
				gold = gift_value
				target = scope:recipient
			}
			ftr_add_political_influence_effect = {
				VALUE = ftr_bribe_influence_gain_value
			}
			trigger_event = ftr_pg_interaction_event.0001
		}
	}
	on_decline = {}
	#auto_accept = yes
	ai_accept = {
		base = -40
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 1
			desc = AI_OPINION_REASON
		}
	}
	ai_will_do = {
		base = -40
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 0.25
			desc = AI_OPINION_REASON
		}
		modifier = {
			add = -20
			scope:actor = {
				has_trait = greedy
			}
		}
		modifier = {
			add = 40
			scope:actor = {
				gold > gift_value
				ftr_political_influence_value < 40
			}
		}
		modifier = {
			add = 10
			scope:actor.liege = scope:recipient
		}
		modifier = {
			add = 10
			scope:actor.liege = scope:recipient.liege
			scope:recipient = { is_councillor = yes }
		}
	}
}

######### Charge ##########
ftr_charge_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	cooldown_against_recipient = { years = 10 }
	desc = ftr_charge_interaction_desc
	ai_targets = {
		ai_recipients = peer_vassals
		max = 2
	}
	ai_frequency = 36
	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			liege = scope:recipient.liege
			is_independent_ruler = no
		}
		scope:recipient = {
			is_independent_ruler = no
		}
		scope:actor.liege = {
			has_imprisonment_reason = scope:recipient
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			is_imprisoned = no
			is_capable_adult = yes
		}
		scope:recipient = {
			is_imprisoned = no
			is_capable_adult = yes
		}
		custom_description = {
			text = "ftr_have_enough_pi_tt"
			value = ftr_charge_cost_value
			ftr_political_influence_value >= ftr_charge_cost_value
		}
	}
	on_accept = {
		scope:actor = {
			ftr_add_political_influence_effect = {
				VALUE = ftr_charge_modify_value
			}
		}
		scope:actor.liege = {
			trigger_event = ftr_pg_interaction_event.0100
		}
	}
	on_decline = {}
	auto_accept = yes
	ai_will_do = {
		base = -20
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = -2
			desc = AI_OPINION_REASON
		}
	}
}
