
########## Buy Favor #########
ftr_buy_favor_interaction = {
	icon = icon_gold
	category = interaction_category_diplomacy
	common_interaction = yes
	ignores_pending_interaction_block = no
	desc = ftr_buy_favor_interaction_desc
	greeting = positive
	notification_text = ftr_buy_favor_interaction_text
	ai_targets = {
		ai_recipients = liege
		ai_recipients = vassals
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 5
	}
	ai_frequency = 24
	cooldown_against_recipient = { years = 5 }

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			gold >= 200
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			NOR = {
				has_usable_hook = scope:recipient
				is_at_war_with = scope:recipient
			}
		}
		scope:recipient = {
			is_adult = yes
			is_imprisoned = no
		}
	}

	send_option = {
		flag = half_price
		localization = "FTR_NEGO_HALF_PRICE"
	}
	send_option = {
		flag = normal_price
		localization = "FTR_NEGO_NORMAL_PRICE"
		starts_enabled = { always = yes	}
	}
	send_option = {
		flag = double_price
		localization = "FTR_NEGO_DOUBLE_PRICE"
	}
	send_option = {
		flag = three_times_price
		localization = "FTR_NEGO_THREE_TIMES_PRICE"
	}
	send_option = {
		flag = four_times_price
		localization = "FTR_NEGO_FOUR_TIMES_PRICE"
	}
	send_option = {
		flag = five_times_price
		localization = "FTR_NEGO_FIVE_TIMES_PRICE"
	}
	send_options_exclusive = yes
	on_accept = {
		scope:actor = {
			send_interface_message = {
				type = event_generic_neutral
				title = ftr_buy_favor_interaction_notification
				left_icon = scope:actor
				right_icon = scope:recipient
				pay_long_term_gold = {
					gold = ftr_buy_favor_cost_value
					target = scope:recipient
				}
				stress_impact = {
					greedy = medium_stress_impact_gain
					generous = medium_stress_impact_loss
				}
				if = {
					limit = {
						NOT = {
							scope:recipient = {
								has_hook = scope:actor
							}
						}
					}
					add_hook = {
						target = scope:recipient
						type = favor_hook
					}
				}
				else = {
					scope:recipient = {
						remove_hook = { target = scope:actor }
					}
				}
			}
		}
	}
	on_decline = {}

	ai_accept = {
		base = -20
		modifier = {
			add = {
				value = 20
				multiply = ftr_payment_cost_level_value
			}
			desc = "FTR_BUY_FAVOR_GOLD_REASON"
		}
		modifier = {
			add = {
				value = scope:recipient.gold
				multiply = -0.1
				min = -100
				max = 100
			}
			desc = "FTR_BUY_FAVOR_FINANCIAL_SITUATION"
		}
		modifier = {
			add = -20
			NOT = { scope:actor.faith = scope:recipient.faith }
			desc = "FTR_BUY_FAVOR_RELIGION_DIFFERENCE"
		}
	}

	ai_will_do = {
		base = 0
		modifier = { # Bad relation
			add = -10
			scope:actor = {
				opinion = {
					target = scope:recipient
					value <= 20
				}
			}
		}
		modifier = {
			add = {
				value = ftr_buy_favor_cost_value
				multiply = -0.05
				min = -50
			}
		}
		modifier = {
			add = 10
			scope:actor = {
				gold >= 300
			}
		}
		modifier = { # Don't need money
			add = -40
			scope:recipient.gold > 100
		}
		modifier = { # Bribe liege
			add = 25
			scope:actor = {
				is_vassal_of = scope:recipient
			}
		}
		modifier = { # Bribe vassals
			add = 25
			scope:recipient = {
				is_vassal_of = scope:actor
			}
		}
	}
}

########## Combat Guidance #########
request_combat_guidance_interaction = {
	icon = icon_combat
	category = interaction_category_friendly
	desc = request_combat_guidance_interaction_desc
	scheme = combat_guide
	send_name = START_SCHEME
	greeting = positive
	notification_text = GUIDE_ME_ON_PROWESS
	cooldown = { years = 2 }
	ignores_pending_interaction_block = no
	ai_targets = {
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = liege
		ai_recipients = guests
		ai_recipients = courtiers
		chance = 0.25
	}
	ai_frequency = 36

	is_shown = {
		NOT = { scope:actor = scope:recipient }
		scope:actor = {
			primary_title.tier > tier_barony
			age >= 12
			OR = {
				has_trait = education_martial
				culture = { has_cultural_parameter = can_duel_rivals }
				has_focus = martial_chivalry_focus
				has_focus = martial_authority_focus
				has_focus = martial_strategy_focus
			}
			prowess < scope:recipient.prowess
		}
	}

	is_valid_showing_failures_only = {
		scope:actor.gold >= 150
		NOT = { scope:actor = { is_at_war_with = scope:recipient } }
		scope:recipient = {
			prowess >= 12
			is_adult = yes
		}
	}

	ai_accept = {
		base = -20
		opinion_modifier = {
			opinion_target = scope:actor
			who = scope:recipient
			multiplier = 0.5
			desc = AI_OPINION_REASON
		}
		modifier = {
			add = 50
			trigger = {
				scope:recipient = {
					gold < 200
				}
			}
			desc = GOLD_REASON
		}
	}

	ai_will_do = {
		base = 0
		modifier = {
			add = -10
			scope:actor = {
				has_trait = shy
			}
		}
		modifier = {
			add = -20
			scope:actor = {
				has_trait = lazy
			}
		}
		modifier = {
			add = -30
			scope:actor = {
				has_trait = content
			}
		}
		modifier = {
			add = -10
			scope:actor = {
				has_trait = craven
			}
		}
		modifier = {
			add = 10
			scope:actor = {
				has_trait = ambitious
			}
		}
		modifier = {
			add = 10
			scope:actor = {
				has_trait = brave
			}
		}
		modifier = {
			add = 10
			scope:actor = {
				has_trait = humble
			}
		}
		modifier = {
			add = 10
			scope:actor = {
				has_trait = diligent
			}
		}
	}

	on_accept = {
		scope:actor = {
			add_opinion = {
				target = scope:recipient
				modifier = ftr_guide_me_on_prowess_opinion
			}
			pay_short_term_gold = {
				gold = 150
				target = scope:recipient
			}
			stress_impact = {
				greedy = medium_stress_impact_gain
				lazy = medium_stress_impact_gain
			}
			send_interface_toast = {
				title = ftr_guidance_accepted_notify_tt
				left_icon = scope:actor					
				right_icon = scope:recipient
				start_scheme = {
					type = combat_guide
					target = scope:recipient
				}
			}
		}
	}
	on_decline = {
		# nothing happen for now
	}
}

########## Religious Debate #########
religious_debate_interaction = {
	category = interaction_category_religion
	common_interaction = yes
	ignores_pending_interaction_block = no
	desc = religious_debate_interaction_desc
	greeting = positive
	notification_text = SOMEONE_START_RELIGIOUS_DEBATE
	ai_targets = {
		ai_recipients = vassals
		ai_recipients = peer_vassals
		ai_recipients = liege
		ai_recipients = neighboring_rulers
		max = 5
	}
	ai_frequency = 36
	cooldown = { years = 2 }

	is_shown = {
		NOT = { scope:actor.faith = scope:recipient.faith }
		scope:actor = { is_imprisoned = no }
		scope:recipient = { is_imprisoned = no }
		NOT = {
			scope:recipient = scope:recipient.faith.religious_head
		}
		scope:actor = {
			OR = {
				has_perk = faithful_perk
				has_trait = zealous
			}
		}
		scope:actor.top_liege = scope:recipient.top_liege
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { is_at_war_with = scope:recipient }
		}
	}

	on_send = {
		scope:recipient = {
			add_character_flag = {
				flag = flag_recent_start_religious_debate
				days = 365
			}
		}
	}

	on_accept = {
		custom_description = {
			text = ftr_start_religious_debate_text
			subject = scope:recipient
			scope:actor = {
				trigger_event = ftr_interaction_event.0001
			}
		}
	}

	on_decline = {}

	#auto_accept = {}
	ai_accept = {
		base = 10
		modifier = {
			add = 50
			trigger = {
				scope:recipient = {
					has_trait = zealous
				}
			}
			desc = "FOR_THE_FAITH"
		}
		modifier = {
			add = 20
			trigger = {
				scope:recipient = {
					learning >= 12
				}
			}
			desc = "FTR_LEARNING_GOOD"
		}
		modifier = {
			add = -50
			trigger = {
				scope:recipient = {
					has_trait = cynical
				}
			}
			desc = "NOT_CARE_ABOUT_RELIGION"
		}
		modifier = {
			add = -100
			trigger = {
				scope:recipient = {
					has_character_flag = flag_recent_start_religious_debate
				}
			}
			desc = "RECENT_RELIGIOUS_DEBATE"
		}
	}

	ai_will_do = {
		base = 10
		modifier = {
			add = -50
			trigger = {
				scope:actor = {
					has_trait = cynical
				}
			}
		}
		modifier = {
			add = 50
			trigger = {
				scope:actor = {
					has_trait = zealous
				}
			}
		}
		modifier = {
			add = -100
			trigger = {
				scope:actor = {
					learning < 12
				}
			}
		}
	}
}
