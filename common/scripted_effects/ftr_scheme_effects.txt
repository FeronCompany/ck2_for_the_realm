
# Coup success effect
ftr_coup_success_effect = {
	$SCHEME$ = {
		save_scope_as = coup_scheme
	}
	scope:coup_scheme = {
		scheme_owner = {
			save_scope_as = owner
		}
		scheme_target_character = {
			save_scope_as = target
		}
		hidden_effect = {
			every_scheme_agent_character = {
				add_opinion = {
					modifier = claimant_faction_member_opinion
					target = scope:owner
				}
				reverse_add_opinion = {
					modifier = claimant_faction_claimant_opinion
					target = scope:owner
				}
			}
		}
	}
	scope:owner = {
		imprison = {
			target = scope:target
			type = dungeon
		}
		add_character_modifier = {
			modifier = ftr_usurper_modifier
			years = 10
		}
		add_dread = 50
	}
	create_title_and_vassal_change = {
		type = usurped
		save_scope_as = change
		add_claim_on_loss = yes
	}
	scope:target = {
		add_opinion = {
			target = scope:owner
			modifier = ftr_success_coup_on_me_opinion
		}
		every_held_title = {
			limit = {
				OR = {
					tier >= tier_kingdom
					tier = primary_title.tier
					AND = {
						tier = tier_county
						this = scope:target.capital_county
					}
				}
			}
			change_title_holder_include_vassals = {
				holder = scope:owner
				change = scope:change
				government_base = scope:target
			}
		}
		every_vassal = {
			change_liege = {
				liege = scope:owner
				change = scope:change
			}
		}
		change_liege = {
			liege = scope:owner
			change = scope:change
		}
	}
	if = {
		limit = {
			scope:target = {
				has_royal_court = yes
			}
			scope:owner = {
				has_royal_court = no
			}
		}
		scope:owner = {
			set_variable = {
				name = previous_holder_grandeur_value
				value = scope:target.court_grandeur_current
			}
		}
	}
	resolve_title_and_vassal_change = scope:change
	scope:owner = {
		if = {
			limit = {
				exists = var:previous_holder_grandeur_value
				has_royal_court = yes
			}
			hidden_effect = {
				change_current_court_grandeur = -100 # Reset to 0
				change_current_court_grandeur = var:previous_holder_grandeur_value
				remove_variable = previous_holder_grandeur_value
			}
			change_current_court_grandeur = massive_court_grandeur_gain # Gain a bonus to Grandeur for winning a claim faction
		}
	}
}

# Coup failure effect
ftr_coup_failure_effect = {
	$SCHEME$ = {
		save_scope_as = coup_scheme
	}
	scope:coup_scheme = {
		scheme_target_character = {
			save_scope_as = target
		}
		every_scheme_agent_character = {
			limit = { NOT = { this = scope:coup_scheme.scheme_owner } }
			scope:target = {
				imprison = {
					target = prev
					type = dungeon
				}
				add_opinion = {
					target = prev
					modifier = ftr_attempted_coup_crime
				}
			}
		}
		scope:target = {
			imprison = {
				target = scope:coup_scheme.scheme_owner
				type = dungeon
			}
			add_opinion = {
				target = scope:coup_scheme.scheme_owner
				modifier = ftr_attempted_coup_crime
			}
		}
		end_scheme = yes
	}
}
