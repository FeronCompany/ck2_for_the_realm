combat_guide = {
	skill = learning
	desc = combat_guide_desc_general
	cooldown = { months = 12 }
	uses_agents = no
	power_per_skill_point = 4
	spymaster_power_per_skill_point = 0	
	resistance_per_skill_point = 0
	spymaster_resistance_per_skill_point = 0
	uses_resistance = no
	minimum_progress_chance = 5
	maximum_progress_chance = 95
	minimum_success = 5
	maximum_success = 95
	icon = icon_scheme_personal
	
	allow = {
		age >= 16
		is_imprisoned = no
		scope:target = {
			age >= 16
			#is_imprisoned = no # TODO this seems unfair to end the scheme for?
		}
	}
	
	valid = {
		scope:target = {
			OR = {
				in_diplomatic_range = scope:owner
			}
		}
	}

	base_success_chance = {
		base = 0
		modifier = {
			add = {
				value = scope:owner.learning
				multiply = 3
			}
			desc = SCHEME_LEARNING_MODIFIER
			scope:owner.learning > 0
		}
		modifier = {
			add = {
				value = 40
				subtract = scope:owner.age
				multiply = 2
			}
			desc = SCHEME_YOUTH_MODIFIER
			scope:owner.age < 30
		}
		modifier = {
			scope:owner = { has_trait = ambitious }
			desc = "SCHEME_TRAIT_AMBITIOUS"
			add = 10
		}
		modifier = {
			scope:owner = { has_trait = diligent }
			desc = "SCHEME_TRAIT_DILIGENT"
			add = 10
		}
		modifier = {
			scope:owner = { has_trait = strong }
			desc = "SCHEME_TRAIT_STRONG"
			add = 30
		}
		modifier = {
			scope:owner = { has_trait = lazy }
			desc = "SCHEME_TRAIT_LAZY"
			add = -20
		}
		modifier = {
			scope:owner = { has_trait = weak }
			desc = "SCHEME_TRAIT_WEAK"
			add = -30
		}
	}

	on_ready = {
		random = {
			chance = scope:scheme.scheme_success_chance

			save_scope_value_as = {
				name = scheme_successful
				value = yes
			}
		}
		scope:owner = {
			if = {
				limit = {
					exists = scope:scheme_successful
				}
				trigger_event = {
					on_action = combat_guide_success
				}
			}
			else = {
				trigger_event = {
					on_action = combat_guide_failure
				}	
			}
		}
	}

	on_invalidated = {
		scheme_target = {
			save_scope_as = target
		}
		scheme_owner = {
			save_scope_as = owner
		}
		if = {
			limit = {
				scope:target = { is_alive = no }
			}
			scope:owner = {
				send_interface_toast = {
					title = combat_guide_invalidated
					custom_tooltip = combat_guide_invalidated_dead_desc
					left_icon = scope:target
				}
			}
		}

		if = {
			limit = {
				scope:target = { 
					NOT = { in_diplomatic_range = scope:owner }
				}
			}
			scope:owner = {
				send_interface_toast = {
					title = combat_guide_invalidated_title
					left_icon = scope:target
					custom_description_no_bullet = {
						object = scope:target
						text = scheme_target_not_in_diplomatic_range
					}
				}
			}
		}
		
		scope:owner = {
			if = {
				limit = {
					is_alive = yes
					scope:target = {
						OR = {
							is_alive = no
							NOT = { in_diplomatic_range = scope:owner }
						}
					}
				}
				set_variable = {
					name = combat_guide_progress
					value = prev.scheme_progress
				}
				trigger_event = combat_guide_outcome.3000
			}
		}
	}

	on_monthly = {
		save_scope_as = scheme
		scheme_target = { save_scope_as = target }
		scheme_owner = {
			save_scope_as = owner
			trigger_event = { on_action = combat_guide_ongoing }
		}
	}
	success_desc = "COMBAT_GUIDE_SUCCESS_DESC"
}
