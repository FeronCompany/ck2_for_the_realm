					}

						add = 30

							has_trait = deceitful
						}
					}
					scope:actor = {
						trigger_event = ftr_interaction_event.0007
					}
					scope:recipient = {
						trigger_event = ftr_interaction_event.0010
					}
					desc = ftr_interaction_event.0003.a.start_lose
