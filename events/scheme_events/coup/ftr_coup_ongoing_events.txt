# Bribe servant
ftr_coup_ongoing.1000 = {
	type = character_event
	title = ftr_coup_ongoing.1000.t
	desc = ftr_coup_ongoing.1000.desc
	theme = intrigue
	left_portrait = {
		character = root
		animation = scheme
	}
	right_portrait = {
		character = scope:agent
		animation = scheme
	}
	
	trigger = {
		NOT = { has_variable = has_ftr_coup_ongoing_1000 }
		scope:scheme = {
			any_scheme_agent = {
				intrigue > 10
				NOT = { root = this }
			}
		}
	}

	immediate = {
		scope:scheme = {
			random_scheme_agent = {
				limit = {
					intrigue > 10
					NOT = { root = this }
				}
				save_scope_as = agent
			}
		}
		set_variable = {	
			name = has_ftr_coup_ongoing_1000
			value = yes
			days = 365
		}
	}

	option = { # Do it
		name = ftr_coup_ongoing.1000.a
		pay_short_term_gold = {
			target = scope:agent
			gold = medium_gold_value
		}
		random_list = {
			10 = {
				modifier = {
					add = scope:agent.intrigue
				}
				custom_description = {
					text = ftr_coup_bribe_servant_suc_text
					trigger_event = {
						id = ftr_coup_ongoing.2000
						days = 5
					}
				}
			}
			10 = {
				# Nothing will happen
			}
		}
		ai_chance = {
			base = 75
		}
	}
	option = { # Forget about it
		name = ftr_coup_ongoing.1000.b
		ai_chance = {
			base = 25
		}
	}
}

					OR = {

			}
		}
	}
