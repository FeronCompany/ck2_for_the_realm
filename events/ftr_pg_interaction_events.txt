ftr_pg_interaction_event.0002 = {
	type = character_event
	title = ftr_pg_interaction_event.0002.t
	desc = ftr_pg_interaction_event.0002.desc
	theme = realm
	left_portrait = {
		character = scope:actor
	}
	right_portrait = {
		character = scope:recipient
	}
	
	option = { # 欣然同意
		name = ftr_pg_interaction_event.0002.a
		scope:actor = {
			send_interface_toast = {
				title = ftr_pg_bribe_cooperation_accept_tt
				left_icon = scope:actor
				right_icon = scope:recipient
				ftr_pg_bribe_cooperation_effect = yes
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				add = -20
				has_trait = just
			}
		}
	}
	option = { # 得加钱
		name = ftr_pg_interaction_event.0002.b
		scope:actor = {
			trigger_event = ftr_pg_interaction_event.0003
		}
		ai_chance = {
			base = 1
			modifier = {
				add = -20
				has_trait = just
			}
			modifier = {
				add = 20
				has_trait = greedy
			}
		}
	}
	option = { # 断然拒绝
		name = ftr_pg_interaction_event.0002.c
		scope:actor = {
			send_interface_toast = {
				title = ftr_pg_bribe_cooperation_reject_tt
				left_icon = scope:actor
				right_icon = scope:recipient
				custom_description = {
					text = "ftr_pg_bribe_cooperation_reject_content_tt"
					object = scope:recipient
					subject = scope:actor
				}
			}
		}
		ai_chance = 5
	}
	option = { # 有请美女
		name = ftr_pg_interaction_event.0002.d
		trigger = {
			scope:recipient = {
				is_male = yes
			}
			scope:actor = {
				liege = scope:recipient
				any_courtier = {
					is_female = yes
					age <= 30
					age >= 16
					OR = {
						has_trait = beauty_good_1
						has_trait = beauty_good_2
						has_trait = beauty_good_3
					}
				}
			}
		}
		scope:actor = {
			random_courtier = {
				limit = {
					is_female = yes
					age <= 30
					age >= 16
					OR = {
						has_trait = beauty_good_1
						has_trait = beauty_good_2
						has_trait = beauty_good_3
					}
				}
				save_scope_as = beauty
			}
			custom_description = {
				text = "ftr_bribe_request_beauty_tt"
				subject = scope:actor
				object = scope:beauty
				trigger_event = ftr_pg_interaction_event.0004
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				add = -20
				OR = {
					has_trait = just
					has_trait = chaste
				}
			}
			modifier = {
				add = 100
				has_trait = lustful
			}
		}
	}
}
